# HTTP → Redirect
server {
    listen 80;
    server_name {{SERVER_NAMES}};
    return 301 https://$host$request_uri;
}

# HTTPS → Static
server {
    listen 443 ssl;
    server_name {{SERVER_NAMES}};

    ssl_certificate     {{SSL_CERT}};
    ssl_certificate_key {{SSL_KEY}};

    if ($host != "{{PRIMARY}}") {
        return 301 https://{{PRIMARY}}$request_uri;
    }

    root {{DEPLOY_PATH}};
    index index.html;

    location / {
        try_files $uri /index.html;
    }
}
